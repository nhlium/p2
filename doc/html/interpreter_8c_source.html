293 /* split str into substrings, separated by char c,
294  which is not white space. The substring has no white space. */
295 static int str_split ( char *str, char c, char *arr[], int max_count )
296 {
297  int count = 0;
298  char *p;
299 
300  for ( p = str; *p != '\0'; p++)
301  {
302  if ( *p == c )
303  {
304  *p = '\0';
305  if (++count > max_count)
306  {
307  put_msg (ERROR, "str_split: more substring than allowed %d\n",
308  max_count);
309  return 0;
310  }
311  }
312  }
313  count++;
314 
315  char str_tmp[11], str_tmp_after_space[11];
316  int i;
317 
318  for ( i = 0, p = str; i < count; i++, p += strlen (p) + 1 )
319  {
320  if ( sscanf (p,"%s%s", str_tmp, str_tmp_after_space) != 1 )
321  {
322  put_msg (ERROR,
323  "str_split: empty string or white space not allowed.\n");
324  for ( --i; i >= 0; i--)
325  {
326  free (arr[i]), arr[i] = NULL;
327  }
328  return 0;
329  }
330  arr[i] = (char *) malloc ( strlen(str_tmp) + 1 );
331  strcpy (arr[i], str_tmp);
332  }
333 
334  return count;
335 }
