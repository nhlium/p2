446 int add_field ( schema_p s, field_desc_p f )
447 {
448  if ( s == NULL )
449  return -1;
450  if ( s->len + f->len > BLOCK_SIZE - PAGE_HEADER_SIZE )
451  {
452  put_msg (ERROR,
453  "schema already has %d bytes, adding %d will exceed limited %d bytes.\n",
454  s->len, f->len, BLOCK_SIZE - PAGE_HEADER_SIZE);
455  return -1;
456  }
457  if ( s->num_fields == 0)
458  {
459  s->first = f;
460  f->offset = 0;
461  }
462  else
463  {
464  s->last->next = f;
465  f->offset = s->len;
466  }
467  s->last = f;
468  s->num_fields++;
469  s->len += f->len;
470  return s->num_fields;
471 }
