395 static pqueue_p release_pqueue (pqueue_p q)
396 {
397  if (q == NULL)
398  return NULL;
399  pq_elm_p nextp = q->first;
400  pq_elm_p p;
401  while (nextp != NULL)
402  {
403  p = nextp; nextp = p->next;
404  free (p);
405  if (nextp == q->last)
406  {
407  if (nextp != q->first)
408  free (nextp); /* beware: already freed if last == first */
409  nextp = NULL;
410  }
411  }
412  free (q);
413  return NULL;

424 /* pg becomes the last in q */
425 static void pq_insert (pqueue_p q, pq_elm_p p)
426 {
427  if (q->first == NULL)
428  { /* q empty */
429  p->prev = p;
430  p->next = p;
431  q->first = p;
432  q->last = p;
433  }
434  else /* q not empty */
435  {
436  p->prev = q->last;
437  p->next = q->first;
438  q->first->prev = p;
439  q->last->next = p;
440  q->last = p;
441  }
442  q->len++;
443  return;
444 }

446 static void pq_remove (pqueue_p q, pq_elm_p p)
447 {
448  if (q->len <= 1)
449  { /* only one element in q */
450  q->first = NULL;
451  q->last = NULL;
452  q->len = 0;
453  }
454  else
455  {
456  p->next->prev = p->prev;
457  p->prev->next = p->next;
458  if (p == q->first)
459  q->first = p->next;
460  if (p == q->last)
461  q->last = p->prev;
462  q->len--;
463  }
464  return;
465 }

613 static fhandle_p get_or_open_tbl_file ( const char *fname )
614 {
615  fhandle_p fh = get_tbl_file (fname);
616 
617  if ( fh != NULL )
618  return fh;
619 
620  if ( num_file_handles == MAX_OPEN_FILES )
621  {
622  put_msg (WARN,
623  "Cannot open file %s because the limit %d of open files has reached.",
624  fname, MAX_OPEN_FILES);
625  return NULL;
626  }
627 
628  int fd = open (fname, O_RDWR, 0);
629  if ( fd == -1 )
630  {
631  /* if the file does not exist, create one */
632  ;
633  if ( (fd = creat (fname, 0600)) == -1 )
634  {
635  put_msg (WARN,
636  "Failed to create file %s.", fname);
637  return NULL;
638  }
639 
640  /* close and open the created file again for read and write */
641  if (close (fd) == -1 || (fd = open (fname, O_RDWR, 0)) == -1)
642  return NULL;
643  }
644 
645  int empty_i = get_empty_fhandle_i ();
646  if ( empty_i == -1 )
647  return NULL;
648 
649  fh = make_fhandle (fname, fd);
650 
651  file_handles[empty_i] = fh;
652  num_file_handles++;
653 
654  return fh;
655 }

826 /* Find an available buffer page, in this order:
827  - unused page,
828  - unpinned page,
829  - unpin a pinned page.

832 {

835  int i;

837  if ( q_pinned->len + q_unpinned->len < NUM_PAGES)
838  {
839  for ( i = 0; pages[i]->block != NULL; i++);
840  pg = pages[i];
841  }

845  pg = pq_dequeue_unpinned (); /* replace an unpinned page */
846  if (pg == NULL)

848  /* If all pages are pinned, unpin the first page
